To access the HTTP API:
{{- if eq .Values.service.type "LoadBalancer" }}

  # Get the external IP/hostname
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "mcp-server-wazuh.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  export SERVICE_HOST=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "mcp-server-wazuh.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].hostname}')

  # Health check
  curl http://${SERVICE_IP:-$SERVICE_HOST}:{{ .Values.service.port }}/health

  # List available tools
  curl -X POST http://${SERVICE_IP:-$SERVICE_HOST}:{{ .Values.service.port }}/mcp \
    -H "Content-Type: application/json" \
    -d '{"jsonrpc":"2.0","id":1,"method":"tools/list","params":{}}'

{{- else if eq .Values.service.type "NodePort" }}

  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "mcp-server-wazuh.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")

  curl http://$NODE_IP:$NODE_PORT/health
